#### 4.3.3 Modelo Físico

```sql
CREATE DATABASE IF NOT EXISTS agilicar
    CHARACTER SET utf8mb4
    COLLATE utf8mb4_unicode_ci;
USE agilicar;

-- =====================================
-- 1. PERFIL
-- =====================================
CREATE TABLE perfil (
    id_perfil   INT          NOT NULL AUTO_INCREMENT,
    nome_perfil VARCHAR(50)  NOT NULL,
    descricao   VARCHAR(255),
    PRIMARY KEY (id_perfil)
) ENGINE=InnoDB;

-- =====================================
-- 2. CLIENTE
-- =====================================
CREATE TABLE cliente (
    id_cliente    INT          NOT NULL AUTO_INCREMENT,
    nome          VARCHAR(100) NOT NULL,
    cpf           VARCHAR(14)  UNIQUE,
    telefone      VARCHAR(20),
    email         VARCHAR(100),
    endereco      VARCHAR(255),
    data_cadastro DATE         NOT NULL DEFAULT (CURRENT_DATE),
    PRIMARY KEY (id_cliente)
) ENGINE=InnoDB;

-- =====================================
-- 3. USUARIO
-- =====================================
CREATE TABLE usuario (
    id_usuario    INT          NOT NULL AUTO_INCREMENT,
    id_perfil     INT          NOT NULL,
    nome          VARCHAR(100) NOT NULL,
    email         VARCHAR(100) NOT NULL UNIQUE,
    senha         VARCHAR(255) NOT NULL,
    ativo         TINYINT(1)   NOT NULL DEFAULT 1,
    data_cadastro DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id_usuario),
    CONSTRAINT fk_usuario_perfil
        FOREIGN KEY (id_perfil) REFERENCES perfil (id_perfil)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB;

-- =====================================
-- 4. VEICULO
-- =====================================
CREATE TABLE veiculo (
    id_veiculo  INT         NOT NULL AUTO_INCREMENT,
    id_cliente  INT         NOT NULL,
    placa       VARCHAR(10) NOT NULL UNIQUE,
    modelo      VARCHAR(100) NOT NULL,
    marca       VARCHAR(100) NOT NULL,
    ano         INT,
    cor         VARCHAR(50),
    chassi      VARCHAR(17) UNIQUE,
    combustivel ENUM('gasolina','etanol','flex','diesel','eletrico','hibrido')
                            DEFAULT 'flex',
    PRIMARY KEY (id_veiculo),
    CONSTRAINT fk_veiculo_cliente
        FOREIGN KEY (id_cliente) REFERENCES cliente (id_cliente)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB;

-- =====================================
-- 5. ORDEM_SERVICO
-- =====================================
CREATE TABLE ordem_servico (
    id_os              INT           NOT NULL AUTO_INCREMENT,
    id_cliente         INT           NOT NULL,
    id_veiculo         INT           NOT NULL,
    id_usuario         INT           NOT NULL,
    data_abertura      DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP,
    status_os          ENUM('aberta','em_execucao','aguardando_peca',
                            'concluida','entregue','cancelada')
                                     NOT NULL DEFAULT 'aberta',
    kilometragem       INT,
    descricao          TEXT,
    descricao_problema TEXT,
    observacoes        TEXT,
    valor_total        DECIMAL(10,2) NOT NULL DEFAULT 0.00,
    PRIMARY KEY (id_os),
    CONSTRAINT fk_os_cliente
        FOREIGN KEY (id_cliente) REFERENCES cliente (id_cliente)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT fk_os_veiculo
        FOREIGN KEY (id_veiculo) REFERENCES veiculo (id_veiculo)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT fk_os_usuario
        FOREIGN KEY (id_usuario) REFERENCES usuario (id_usuario)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB;

-- =====================================
-- 6. SERVICO
-- =====================================
CREATE TABLE servico (
    id_servico   INT           NOT NULL AUTO_INCREMENT,
    nome_servico VARCHAR(100)  NOT NULL,
    descricao    TEXT,
    valor        DECIMAL(10,2) NOT NULL DEFAULT 0.00,
    PRIMARY KEY (id_servico)
) ENGINE=InnoDB;

-- =====================================
-- 7. OS_SERVICO (associativa N:M)
-- =====================================
CREATE TABLE os_servico (
    id_os           INT           NOT NULL,
    id_servico      INT           NOT NULL,
    id_tecnico      INT           NOT NULL,
    quantidade      INT           NOT NULL DEFAULT 1,
    valor_cobrado   DECIMAL(10,2) NOT NULL DEFAULT 0.00,
    inicio_execucao DATETIME,
    fim_execucao    DATETIME,
    PRIMARY KEY (id_os, id_servico),
    CONSTRAINT fk_osservico_os
        FOREIGN KEY (id_os)      REFERENCES ordem_servico (id_os)
        ON UPDATE CASCADE ON DELETE CASCADE,
    CONSTRAINT fk_osservico_servico
        FOREIGN KEY (id_servico) REFERENCES servico (id_servico)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT fk_osservico_tecnico
        FOREIGN KEY (id_tecnico) REFERENCES usuario (id_usuario)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB;

-- =====================================
-- 8. ORCAMENTO
-- =====================================
CREATE TABLE orcamento (
    id_orcamento     INT           NOT NULL AUTO_INCREMENT,
    id_os            INT           NOT NULL UNIQUE,
    data_orcamento   DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP,
    valor_total      DECIMAL(10,2) NOT NULL DEFAULT 0.00,
    status_orcamento ENUM('pendente','aprovado','reprovado','cancelado')
                                   NOT NULL DEFAULT 'pendente',
    observacoes      TEXT,
    PRIMARY KEY (id_orcamento),
    CONSTRAINT fk_orcamento_os
        FOREIGN KEY (id_os) REFERENCES ordem_servico (id_os)
        ON UPDATE CASCADE ON DELETE CASCADE
) ENGINE=InnoDB;

-- =====================================
-- 9. PECA
-- =====================================
CREATE TABLE peca (
    id_peca        INT           NOT NULL AUTO_INCREMENT,
    nome_peca      VARCHAR(100)  NOT NULL,
    descricao      TEXT,
    codigo         VARCHAR(50)   UNIQUE,
    fornecedor     VARCHAR(100),
    quantidade     INT           NOT NULL DEFAULT 0,
    valor_unitario DECIMAL(10,2) NOT NULL DEFAULT 0.00,
    PRIMARY KEY (id_peca)
) ENGINE=InnoDB;

-- =====================================
-- 10. ESTOQUE
-- =====================================
CREATE TABLE estoque (
    id_estoque            INT          NOT NULL AUTO_INCREMENT,
    id_peca               INT          NOT NULL UNIQUE,
    quantidade_disponivel INT          NOT NULL DEFAULT 0,
    estoque_minimo        INT          NOT NULL DEFAULT 2,
    localizacao           VARCHAR(100),
    PRIMARY KEY (id_estoque),
    CONSTRAINT fk_estoque_peca
        FOREIGN KEY (id_peca) REFERENCES peca (id_peca)
        ON UPDATE CASCADE ON DELETE CASCADE
) ENGINE=InnoDB;

-- =====================================
-- 11. OS_PECA (associativa N:M)
-- =====================================
CREATE TABLE os_peca (
    id_os          INT           NOT NULL,
    id_peca        INT           NOT NULL,
    quantidade     INT           NOT NULL DEFAULT 1,
    valor_unitario DECIMAL(10,2) NOT NULL DEFAULT 0.00,
    PRIMARY KEY (id_os, id_peca),
    CONSTRAINT fk_ospeca_os
        FOREIGN KEY (id_os)   REFERENCES ordem_servico (id_os)
        ON UPDATE CASCADE ON DELETE CASCADE,
    CONSTRAINT fk_ospeca_peca
        FOREIGN KEY (id_peca) REFERENCES peca (id_peca)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB;

-- =====================================
-- 12. REGISTRO_FOTOGRAFICO
-- =====================================
CREATE TABLE registro_fotografico (
    id_foto       INT          NOT NULL AUTO_INCREMENT,
    id_os         INT          NOT NULL,
    imagem        VARCHAR(255) NOT NULL,
    descricao     TEXT,
    momento       ENUM('antes','durante','depois') NOT NULL DEFAULT 'durante',
    data_registro DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id_foto),
    CONSTRAINT fk_foto_os
        FOREIGN KEY (id_os) REFERENCES ordem_servico (id_os)
        ON UPDATE CASCADE ON DELETE CASCADE
) ENGINE=InnoDB;

-- =====================================
-- 13. NOTIFICACAO
-- =====================================
CREATE TABLE notificacao (
    id_notificacao INT  NOT NULL AUTO_INCREMENT,
    id_os          INT  NOT NULL,
    mensagem       TEXT NOT NULL,
    data_envio     DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    status_envio   ENUM('pendente','enviado','falha')          NOT NULL DEFAULT 'pendente',
    tipo           ENUM('status','orcamento','entrega','aviso') NOT NULL DEFAULT 'status',
    PRIMARY KEY (id_notificacao),
    CONSTRAINT fk_notificacao_os
        FOREIGN KEY (id_os) REFERENCES ordem_servico (id_os)
        ON UPDATE CASCADE ON DELETE CASCADE
) ENGINE=InnoDB;

-- =====================================
-- 14. ENTREGA
-- =====================================
CREATE TABLE entrega (
    id_entrega         INT        NOT NULL AUTO_INCREMENT,
    id_os              INT        NOT NULL UNIQUE,
    data_entrega       DATETIME   NOT NULL DEFAULT CURRENT_TIMESTAMP,
    confirmacao        TINYINT(1) NOT NULL DEFAULT 0,
    kilometragem_saida INT,
    observacao         TEXT,
    assinado_cliente   TINYINT(1) NOT NULL DEFAULT 0,
    PRIMARY KEY (id_entrega),
    CONSTRAINT fk_entrega_os
        FOREIGN KEY (id_os) REFERENCES ordem_servico (id_os)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB;

-- =====================================
-- DADOS INICIAIS
-- =====================================
INSERT INTO perfil (nome_perfil, descricao) VALUES
    ('gestor',    'Acesso completo ao sistema'),
    ('atendente', 'Abertura e acompanhamento de OS'),
    ('tecnico',   'Execução de serviços'),
    ('cliente',   'Consulta de status e histórico');

-- =====================================
-- VIEWS AUXILIARES
-- =====================================

-- Resumo completo de cada OS
CREATE OR REPLACE VIEW vw_resumo_os AS
    SELECT
        os.id_os,
        os.data_abertura,
        os.status_os,
        os.kilometragem,
        os.descricao_problema,
        os.valor_total,
        c.id_cliente,
        c.nome     AS nome_cliente,
        c.telefone AS telefone_cliente,
        v.placa,
        v.marca,
        v.modelo,
        v.ano,
        u.nome     AS nome_responsavel
    FROM ordem_servico os
    INNER JOIN cliente c ON os.id_cliente = c.id_cliente
    INNER JOIN veiculo v ON os.id_veiculo = v.id_veiculo
    INNER JOIN usuario u ON os.id_usuario = u.id_usuario;

-- Peças com estoque abaixo do mínimo
CREATE OR REPLACE VIEW vw_estoque_critico AS
    SELECT
        p.id_peca,
        p.nome_peca,
        p.codigo,
        p.fornecedor,
        e.quantidade_disponivel,
        e.estoque_minimo,
        e.localizacao
    FROM peca p
    INNER JOIN estoque e ON p.id_peca = e.id_peca
    WHERE e.quantidade_disponivel <= e.estoque_minimo;
```
